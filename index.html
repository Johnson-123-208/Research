import requests
from google.oauth2.credentials import Credentials

# Authentication setup
def get_google_fit_data(access_token):
    headers = {
        'Authorization': f'Bearer {access_token}',
        'Content-Type': 'application/json'
    }
    
    # Get step count data
    url = 'https://www.googleapis.com/fitness/v1/users/me/dataSources'
    response = requests.get(url, headers=headers)
    
    # Get aggregated data
    agg_url = 'https://www.googleapis.com/fitness/v1/users/me/dataset:aggregate'
    payload = {
        "aggregateBy": [{
            "dataTypeName": "com.google.step_count.delta",
            "dataSourceId": "derived:com.google.step_count.delta:com.google.android.gms:estimated_steps"
        }],
        "bucketByTime": {"durationMillis": 86400000},  # 1 day
        "startTimeMillis": 1640995200000,
        "endTimeMillis": 1641081600000
    }
    
    agg_response = requests.post(agg_url, json=payload, headers=headers)
    return agg_response.json()

# Usage
data = get_google_fit_data('YOUR_ACCESS_TOKEN')

Apple HealthKit (iOS)

Setup Process:
1. Enable HealthKit capability in Xcode
2. Add usage description in Info.plist
3. Request permissions from user
4. Query health data 

import HealthKit

class HealthKitManager {
    let healthStore = HKHealthStore()
    
    func requestPermissions() {
        let healthTypes: Set = [
            HKObjectType.quantityType(forIdentifier: .heartRate)!,
            HKObjectType.quantityType(forIdentifier: .stepCount)!,
            HKObjectType.categoryType(forIdentifier: .sleepAnalysis)!
        ]
        
        healthStore.requestAuthorization(toShare: nil, read: healthTypes) { success, error in
            if success {
                self.fetchHeartRateData()
            }
        }
    }
    
    func fetchHeartRateData() {
        let heartRateType = HKQuantityType.quantityType(forIdentifier: .heartRate)!
        let query = HKSampleQuery(sampleType: heartRateType,
                                 predicate: nil,
                                 limit: 100,
                                 sortDescriptors: nil) { query, samples, error in
            guard let heartRateSamples = samples as? [HKQuantitySample] else { return }
            
            for sample in heartRateSamples {
                let heartRate = sample.quantity.doubleValue(for: HKUnit.count().unitDivided(by: .minute()))
                print("Heart Rate: \(heartRate) BPM at \(sample.startDate)")
            }
        }
        healthStore.execute(query)
    }
}

Samsung Health API (Android)

Setup Process:
1. Register your app with Samsung Health SDK
2. Add required permissions in AndroidManifest.xml
3. Connect to HealthDataStore
4. Query health data (steps, HR, sleep) 

import com.samsung.android.sdk.healthdata.*
                
                class SamsungHealthManager(context: Context) {
                    private val mStore: HealthDataStore = HealthDataStore(context, object : HealthDataStore.ConnectionListener {
                        override fun onConnected() {
                            println("Samsung Health connected")
                        }
                        override fun onConnectionFailed(error: HealthConnectionErrorResult) {
                            println("Connection failed: ${error.message}")
                        }
                        override fun onDisconnected() {}
                    })
                
                    init {
                        mStore.connectService()
                    }
                
                    fun readSteps() {
                        val dataType = HealthConstants.StepCount.HEALTH_DATA_TYPE
                        val resolver = HealthDataResolver(mStore, null)
                        val request = HealthDataResolver.ReadRequest.Builder()
                            .setDataType(dataType)
                            .build()
                        val result = resolver.read(request)
                        result.await().use { res ->
                            res.iterator().forEach {
                                println("Steps: ${it.getInt(HealthConstants.StepCount.COUNT)}")
                            }
                        }
                    }
                }

import requests
from datetime import datetime, timedelta

class OuraAPI:
    def __init__(self, access_token):
        self.access_token = access_token
        self.base_url = 'https://api.ouraring.com/v2'
        self.headers = {
            'Authorization': f'Bearer {access_token}',
            'Content-Type': 'application/json'
        }
    
    def get_sleep_data(self, start_date=None, end_date=None):
        if not start_date:
            start_date = (datetime.now() - timedelta(days=7)).strftime('%Y-%m-%d')
        if not end_date:
            end_date = datetime.now().strftime('%Y-%m-%d')
        
        url = f'{self.base_url}/usercollection/sleep'
        params = {
            'start_date': start_date,
            'end_date': end_date
        }
        response = requests.get(url, headers=self.headers, params=params)
        return response.json()
    
    def get_readiness_data(self, start_date=None, end_date=None):
        if not start_date:
            start_date = (datetime.now() - timedelta(days=7)).strftime('%Y-%m-%d')
        
        url = f'{self.base_url}/usercollection/daily_readiness'
        params = {
            'start_date': start_date,
            'end_date': end_date
        }
        response = requests.get(url, headers=self.headers, params=params)
        return response.json()
    
    def get_hrv_data(self, start_date=None, end_date=None):
        url = f'{self.base_url}/usercollection/heartrate'
        params = {
            'start_datetime': f'{start_date}T00:00:00.000Z',
            'end_datetime': f'{end_date}T23:59:59.999Z'
        }
        response = requests.get(url, headers=self.headers, params=params)
        return response.json()

# Usage
oura = OuraAPI('YOUR_ACCESS_TOKEN')

Garmin Connect IQ

Getting API Access:
1. Register at developer.garmin.com
2. Create application and get credentials
3. Implement OAuth 1.0a flow
4. Access wellness and activity data 

import requests
import hashlib
import hmac
import base64
from urllib.parse import quote
import time
import random

class GarminConnectAPI:
    def __init__(self, consumer_key, consumer_secret):
        self.consumer_key = consumer_key
        self.consumer_secret = consumer_secret
        self.base_url = 'https://apis.garmin.com/wellness-api/rest'
    
    def generate_oauth_signature(self, method, url, params, token_secret=''):
        # OAuth 1.0a signature generation
        sorted_params = sorted(params.items())
        param_string = '&'.join([f'{k}={v}' for k, v in sorted_params])
        
        base_string = f'{method}&{quote(url, safe="")}&{quote(param_string, safe="")}'
        signing_key = f'{quote(self.consumer_secret, safe="")}&{quote(token_secret, safe="")}'
        
        signature = base64.b64encode(
            hmac.new(signing_key.encode(), base_string.encode(), hashlib.sha1).digest()
        ).decode()
        
        return signature
    
    def get_user_summary(self, access_token, token_secret, date):
        url = f'{self.base_url}/dailies'
        
        oauth_params = {
            'oauth_consumer_key': self.consumer_key,
            'oauth_token': access_token,
            'oauth_signature_method': 'HMAC-SHA1',
            'oauth_timestamp': str(int(time.time())),
            'oauth_nonce': str(random.randint(1000000, 9999999)),
            'oauth_version': '1.0',
            'calendarDate': date
        }
        
        signature = self.generate_oauth_signature('GET', url, oauth_params, token_secret)
        oauth_params['oauth_signature'] = signature
        
        response = requests.get(url, params=oauth_params)
        return response.json()
    
    def get_activities(self, access_token, token_secret, start_date, end_date):
        url = f'{self.base_url}/activities'
        
        oauth_params = {
            'oauth_consumer_key': self.consumer_key,
            'oauth_token': access_token,
            'oauth_signature_method': 'HMAC-SHA1',
            'oauth_timestamp': str(int(time.time())),
            'oauth_nonce': str(random.randint(1000000, 9999999)),
            'oauth_version': '1.0',
            'startDate': start_date,
            'endDate': end_date
        }
        
        signature = self.generate_oauth_signature('GET', url, oauth_params, token_secret)
        oauth_params['oauth_signature'] = signature
        
        response = requests.get(url, params=oauth_params)
        return response.json()

# Usage
garmin = GarminConnectAPI('YOUR_CONSUMER_KEY', 'YOUR_CONSUMER_SECRET')
summary = garmin.get_user_summary('ACCESS_TOKEN', 'TOKEN_SECRET', '2024-01-01')

import requests
import base64
from datetime import datetime

class PolarAccessLinkAPI:
    def __init__(self, client_id, client_secret):
        self.client_id = client_id
        self.client_secret = client_secret
        self.base_url = 'https://www.polaraccesslink.com/v3'
    
    def get_access_token(self, authorization_code):
        url = 'https://polarremote.com/v2/oauth2/token'
        
        auth_header = base64.b64encode(f'{self.client_id}:{self.client_secret}'.encode()).decode()
        headers = {
            'Authorization': f'Basic {auth_header}',
            'Content-Type': 'application/x-www-form-urlencoded'
        }
        
        data = {
            'grant_type': 'authorization_code',
            'code': authorization_code
        }
        
        response = requests.post(url, headers=headers, data=data)
        return response.json()
    
    def register_user(self, access_token, member_id):
        url = f'{self.base_url}/users'
        headers = {
            'Authorization': f'Bearer {access_token}',
            'Content-Type': 'application/json'
        }
        data = {'member-id': member_id}
        
        response = requests.post(url, headers=headers, json=data)
        return response.status_code == 200
    
    def get_available_data(self, access_token, user_id):
        url = f'{self.base_url}/users/{user_id}/exercise-transactions'
        headers = {'Authorization': f'Bearer {access_token}'}
        
        response = requests.get(url, headers=headers)
        return response.json()
    
    def get_exercise_summary(self, access_token, user_id, transaction_id):
        url = f'{self.base_url}/users/{user_id}/exercise-transactions/{transaction_id}'
        headers = {'Authorization': f'Bearer {access_token}'}
        
        response = requests.get(url, headers=headers)
        return response.json()
    
    def get_daily_activity(self, access_token, user_id, date):
        url = f'{self.base_url}/users/{user_id}/activity-transactions'
        headers = {'Authorization': f'Bearer {access_token}'}
        
        response = requests.get(url, headers=headers)
        return response.json()

# Usage
polar = PolarAccessLinkAPI('YOUR_CLIENT_ID', 'YOUR_CLIENT_SECRET')
token_data = polar.get_access_token('AUTHORIZATION_CODE')
access_token = token_data['access_token']
user_registered = polar.register_user(access_token, 'MEMBER_ID')
